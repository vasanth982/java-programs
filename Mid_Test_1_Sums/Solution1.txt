import java.util.HashSet;
class Solution1{
public boolean containsNearbydublicate(int[] nums,int k){
HashSet<integer>set=new HashSet<>();
for(int i=0;i<nums.length;i++){
if(set.contain(nums[i])){
return true;
}
set.add(nums[i]);
if(set.size()>k){
set.remove(nums[i-k]);
}
}
return false;
}
public static void main(String[] args){
Solution1 obj = new Solution1();
int[] nums = {1,2,3,1};
int k = 3;
boolean result = obj.containsNearbyduplicates(nums,k);
System.out.println("Result:" +result);
}
}
